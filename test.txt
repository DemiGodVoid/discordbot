import discord
import json
import os

intents = discord.Intents.default()
intents.message_content = True
client = discord.Client(intents=intents)

TOKEN = input("Please enter your Discord bot token: ")

GIFS_FILE = "gifs.json"

# Load GIFs from file
def load_gifs():
    if os.path.exists(GIFS_FILE):
        with open(GIFS_FILE, "r") as f:
            return json.load(f)
    return {}

# Save GIFs to file
def save_gifs(gifs):
    with open(GIFS_FILE, "w") as f:
        json.dump(gifs, f, indent=4)

gifs = load_gifs()  # Load existing GIFs

@client.event
async def on_ready():
    print(f'We have logged in as {client.user}')

@client.event
async def on_message(message):
    if message.author == client.user:
        return

    if message.content.startswith("!gif "):
        keyword = message.content[len("!gif "):].strip().lower()
        if keyword in gifs:
            await message.channel.send(gifs[keyword])  # Send the stored GIF URL
        else:
            await message.channel.send(f"No GIF found for '{keyword}'. Use `!addgif {keyword} <gif_url>` to add one.")

    if message.content.startswith("!addgif "):
        parts = message.content[len("!addgif "):].split(" ", 1)
        if len(parts) == 2:
            keyword, gif_url = parts[0].strip().lower(), parts[1].strip()
            gifs[keyword] = gif_url
            save_gifs(gifs)  # Save the new GIF
            await message.channel.send(f"GIF added for '{keyword}': {gif_url}")
        else:
            await message.channel.send("Invalid format. Use: `!addgif <keyword> <gif_url>`")

client.run(TOKEN)
